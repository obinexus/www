Usage Examples

Basic Trace Command
$ diram.exe --trace ./path/to/lib/*.so --detach
# Traces all .so files in detached mode
# Outputs to diram.log with QA metrics
C Integration
#include <diram.h>

int main() {
  // Initialize DIRAM tracer
  diram_init();
  diram_set_target("95.4");
  
  // Your code here
  void* ptr = malloc(1024);
  
  // DIRAM tracks automatically
  diram_report();
  return 0;
}
Makefile Integration
CFLAGS += -g -O0
LDFLAGS += -ldiram

# Enable DIRAM debugging
debug: CFLAGS += -DDIRAM_TRACE
debug: $(TARGET)
  diram.exe --trace $(TARGET) --report
GDB Script
# .gdbinit for DIRAM
source diram.gdb
diram-attach

# Set breakpoints on errors
diram-break-on-leak
diram-break-on-overflow

# Run with QA metrics
diram-run --qa-target 95.4
